# ------------------------------------------------------------
# Original Tag Inheritance (Dynamic Civil Wars)
# ------------------------------------------------------------
# These triggers allow a civil war tag to identify its path 
# based on purely being in a civil war under a specific ideology.
# They inherit focuses from the parent via base game mechanics,
# so they just need the ideology check as a tie-breaker.

is_generic_fascist_civil_war_path = {
    is_subject = no
    has_government = fascism
    has_civil_war = yes
}

is_generic_communist_civil_war_path = {
    is_subject = no
    has_government = communism
    has_civil_war = yes
}

is_generic_democratic_civil_war_path = {
    is_subject = no
    has_government = democratic
    has_civil_war = yes
}

is_generic_neutrality_civil_war_path = {
    is_subject = no
    has_government = neutrality
    has_civil_war = yes
}

# ------------------------------------------------------------
# Generic Ideology Fallback (For any random releasable)
# ------------------------------------------------------------
# If a minor colony, puppet or breakaway is released, this determines 
# its basic behavior strictly off its government.

is_generic_releasable_fascist_path = {
    is_major = no
    has_government = fascism
    NOT = { has_civil_war = yes }
    NOT = { has_focus_tree = default } # Only applies if we're not running a major tree
}

is_generic_releasable_communist_path = {
    is_major = no
    has_government = communism
    NOT = { has_civil_war = yes }
    NOT = { has_focus_tree = default }
}

is_generic_releasable_democratic_path = {
    is_major = no
    has_government = democratic
    NOT = { has_civil_war = yes }
    NOT = { has_focus_tree = default }
}

is_generic_releasable_neutral_path = {
